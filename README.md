<h1> Online Pizza Ordering </h1>

 <h3>Technologies Used:</h3>
 <ol>
    <li>Backend: Spring Boot, Rest API</li>
     <li>Database: MySQL</li>
     <li>Frontend: ReactJS</li>
   </ol>
<!-- <br> -->
 <h3>Working flow:</h3>
  <ol>
    <li>Registration</li>
    <li>Login</li>
    <li>view food item </li>
    <li>add into Cart</li>
    <li>Proceed for Checkout</li>
    <li>Proceed for payment </li>
    <li> Order placed!!</li>
   </ol>
